Anima — commands (paths are relative to repository root)

=== Bootstrap (recommended first run) ===
  python scripts/bootstrap.py
  python scripts/bootstrap.py --skip-train
  python scripts/bootstrap.py --skip-tests

=== Install ===
  pip install -e ".[dev]"
  cd dashboard && npm install && cp .env.example .env

=== API (port 8010) ===
  anima api --port 8010
  python -m uvicorn api.server:app --host 127.0.0.1 --port 8010

=== Dashboard ===
  cd dashboard && npm run dev
  open http://127.0.0.1:5173

=== Train probes ===
  python scripts/download_narratives_minimal.py
  export NARRATIVES_ROOT=./data/narratives_minimal
  python scripts/train_all_probes.py
  anima train-text --model distilgpt2 --max-samples 1000
  anima train --narratives-root ./data/narratives_minimal --model distilgpt2

=== Benchmarks ===
  python scripts/setup_benchmarks.py
  anima benchmark --model hf-internal-testing/tiny-random-gpt2 --tiers internal,external,external_text,external_guard

=== Tests ===
  python -m pytest -q -k "not distilgpt2"
  RUN_HF_TESTS=1 python -m pytest -q

=== Docker ===
  docker compose build && docker compose up
